# E2E Tests — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Test mlflow-dynamodbstore through the full HTTP stack by running a real MLflow server process and exercising it via the MLflow client SDK, catching serialization bugs (Decimal→protobuf), missing fields, and response format issues that unit tests miss.

**Architecture:** A session-scoped fixture starts a moto server (DynamoDB backend), then launches `mlflow server` as a subprocess pointing at it. Tests use `MlflowClient` to make real HTTP requests. Each test module resets state between tests.

**Tech Stack:** pytest, moto server, subprocess (mlflow server), `mlflow.MlflowClient`, `mlflow` top-level API.

**Depends on:** Current `enhance/e2e-tests` branch (compatibility tests already moved to `tests/unit/`).

# tests/e2e/conftest.py
"""E2E test fixtures: moto server + mlflow server subprocess."""

import os
import subprocess
import sys
import time

import pytest
import requests
from mlflow import MlflowClient
from moto.server import ThreadedMotoServer

pytestmark = pytest.mark.e2e


@pytest.fixture(scope="session", autouse=True)
def aws_credentials():
    """Set fake AWS credentials for the moto server."""
    os.environ["AWS_ACCESS_KEY_ID"] = "testing"
    os.environ["AWS_SECRET_ACCESS_KEY"] = "testing"
    os.environ["AWS_SECURITY_TOKEN"] = "testing"
    os.environ["AWS_SESSION_TOKEN"] = "testing"
    os.environ["AWS_DEFAULT_REGION"] = "us-east-1"


@pytest.fixture(scope="session")
def moto_endpoint(aws_credentials):
    """Start a moto server for DynamoDB."""
    server = ThreadedMotoServer(port=0)
    server.start()
    host, port = server.get_host_and_port()
    yield f"http://localhost:{port}"
    server.stop()


@pytest.fixture(scope="session")
def mlflow_server(moto_endpoint):
    """Start an mlflow server subprocess backed by moto DynamoDB.

    Returns the tracking URI (http://localhost:<port>).
    """
    env = os.environ.copy()
    env["MLFLOW_FLASK_SERVER_SECRET_KEY"] = "test-secret-key"

    store_uri = f"dynamodb://{moto_endpoint}/e2e-test-table"

    proc = subprocess.Popen(
        [
            sys.executable,
            "-m",
            "mlflow",
            "server",
            "--backend-store-uri",
            store_uri,
            "--default-artifact-root",
            "/tmp/mlflow-e2e-artifacts",
            "--host",
            "127.0.0.1",
            "--port",
            "0",  # Let OS pick a free port
            "--workers",
            "1",
        ],
        env=env,
        stdout=subprocess.PIPE,
        stderr=subprocess.STDOUT,
        text=True,
    )

    # Wait for server to be ready by polling
    # Read stdout to find the actual port
    tracking_uri = None
    deadline = time.time() + 30
    while time.time() < deadline:
        line = proc.stdout.readline()
        if not line:
            time.sleep(0.1)
            continue
        # Uvicorn prints: "Uvicorn running on http://127.0.0.1:PORT"
        if "Uvicorn running on" in line:
            # Extract URL
            tracking_uri = line.split("Uvicorn running on ")[-1].strip().rstrip("(Press CTRL+C to quit)").strip()
            break

    if tracking_uri is None:
        proc.kill()
        raise RuntimeError("MLflow server did not start within 30s")

    # Wait for server to respond to health check
    deadline = time.time() + 15
    while time.time() < deadline:
        try:
            resp = requests.get(f"{tracking_uri}/health", timeout=2)
            if resp.status_code == 200:
                break
        except requests.ConnectionError:
            time.sleep(0.5)
    else:
        proc.kill()
        raise RuntimeError("MLflow server health check failed")

    yield tracking_uri

    proc.terminate()
    proc.wait(timeout=10)


@pytest.fixture(scope="session")
def client(mlflow_server) -> MlflowClient:
    """Return an MlflowClient pointed at the e2e server."""
    return MlflowClient(tracking_uri=mlflow_server)
```

Note: `--port 0` may not work with mlflow. If it doesn't, use a fixed high port like `15123` or find a free port with `socket`. The implementer should verify this and adjust.

import pytest
from mlflow import MlflowClient
from mlflow.exceptions import MlflowException

pytestmark = pytest.mark.e2e

    def test_default_experiment_exists(self, client: MlflowClient):
        """MLflow UI loads default experiment on startup."""

        assert exp.name == "Default"
        assert exp.lifecycle_stage == "active"

    def test_create_experiment(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-test-exp")
        exp = client.get_experiment(exp_id)
        assert exp.name == "e2e-test-exp"
        assert exp.experiment_id == exp_id

        name = "e2e-by-name"
        exp_id = client.create_experiment(name)
        exp = client.get_experiment_by_name(name)
        assert exp is not None
        assert exp.experiment_id == exp_id

    def test_search_experiments(self, client: MlflowClient):
        """This is the exact call the MLflow UI makes on page load."""
        experiments = client.search_experiments(
            max_results=25,
            order_by=["last_update_time DESC"],
        )
        assert len(experiments) >= 1  # At least Default

    def test_search_experiments_with_filter(self, client: MlflowClient):
        client.create_experiment("e2e-filter-test")
        experiments = client.search_experiments(
            filter_string="name = 'e2e-filter-test'",
        )
        assert len(experiments) == 1
        assert experiments[0].name == "e2e-filter-test"

    def test_delete_and_restore_experiment(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-delete-exp")
        client.delete_experiment(exp_id)
        exp = client.get_experiment(exp_id)
        assert exp.lifecycle_stage == "deleted"

        client.restore_experiment(exp_id)
        exp = client.get_experiment(exp_id)
        assert exp.lifecycle_stage == "active"

    def test_set_experiment_tag(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-tag-exp")
        client.set_experiment_tag(exp_id, "team", "ml-platform")
        exp = client.get_experiment(exp_id)

        exp_id = client.create_experiment("e2e-rename-old")
        client.rename_experiment(exp_id, "e2e-rename-new")

# tests/e2e/test_runs.py
"""E2E tests for run operations via MLflow client SDK."""

import pytest
import mlflow
from mlflow import MlflowClient
from mlflow.entities import Metric, Param, RunTag, ViewType

pytestmark = pytest.mark.e2e


@pytest.fixture
def experiment_id(client: MlflowClient):
    """Create a fresh experiment for run tests."""
    return client.create_experiment(f"e2e-runs-{id(client)}")


class TestRuns:
    def test_create_run(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        assert run.info.experiment_id == experiment_id
        assert run.info.status == "RUNNING"

    def test_log_metrics(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        client.log_metric(run.info.run_id, "accuracy", 0.95)
        client.log_metric(run.info.run_id, "loss", 0.05)

        fetched = client.get_run(run.info.run_id)
        metrics = {m.key: m.value for m in fetched.data.metrics}
        assert metrics["accuracy"] == 0.95
        assert metrics["loss"] == 0.05

    def test_log_params(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        client.log_param(run.info.run_id, "learning_rate", "0.01")
        client.log_param(run.info.run_id, "epochs", "100")

        fetched = client.get_run(run.info.run_id)
        params = {p.key: p.value for p in fetched.data.params}
        assert params["learning_rate"] == "0.01"
        assert params["epochs"] == "100"

    def test_set_tag(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        client.set_tag(run.info.run_id, "model_type", "xgboost")

        fetched = client.get_run(run.info.run_id)
        assert fetched.data.tags["model_type"] == "xgboost"

    def test_log_batch(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        client.log_batch(
            run.info.run_id,
            metrics=[Metric("m1", 1.0, 0, 0), Metric("m2", 2.0, 0, 0)],
            params=[Param("p1", "v1")],
            tags=[RunTag("t1", "v1")],
        )

        fetched = client.get_run(run.info.run_id)
        assert len(fetched.data.metrics) >= 2
        assert len(fetched.data.params) >= 1

    def test_update_run(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        client.set_terminated(run.info.run_id, status="FINISHED")

        fetched = client.get_run(run.info.run_id)
        assert fetched.info.status == "FINISHED"
        assert fetched.info.end_time is not None

    def test_delete_and_restore_run(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        client.delete_run(run.info.run_id)
        fetched = client.get_run(run.info.run_id)
        assert fetched.info.lifecycle_stage == "deleted"

        client.restore_run(run.info.run_id)
        fetched = client.get_run(run.info.run_id)
        assert fetched.info.lifecycle_stage == "active"

    def test_get_metric_history(self, client: MlflowClient, experiment_id):
        run = client.create_run(experiment_id)
        for i in range(5):
            client.log_metric(run.info.run_id, "loss", 1.0 - i * 0.1, step=i)
        history = client.get_metric_history(run.info.run_id, "loss")
        assert len(history) == 5

    def test_mlflow_fluent_api(self, mlflow_server, client: MlflowClient):
        """Test the high-level mlflow.* API that most users use."""
        mlflow.set_tracking_uri(mlflow_server)
        mlflow.set_experiment("e2e-fluent-test")

        with mlflow.start_run() as run:
            mlflow.log_param("lr", "0.01")
            mlflow.log_metric("acc", 0.99)
            mlflow.set_tag("framework", "pytorch")

        fetched = client.get_run(run.info.run_id)
        assert fetched.data.params["lr"] == "0.01"
        assert fetched.data.tags["framework"] == "pytorch"
```

"""E2E tests for search operations — these mirror the exact HTTP calls the MLflow UI makes."""

import pytest
import mlflow
from mlflow import MlflowClient
from mlflow.entities import ViewType

pytestmark = pytest.mark.e2e


class TestSearchRuns:
    def test_search_runs_basic(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-search-runs")
        run = client.create_run(exp_id)
        client.log_metric(run.info.run_id, "acc", 0.9)
        client.set_terminated(run.info.run_id, "FINISHED")

        runs = client.search_runs(
            experiment_ids=[exp_id],
            max_results=100,
        )
        assert len(runs) >= 1
        assert runs[0].info.run_id == run.info.run_id

    def test_search_runs_with_filter(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-search-filter")
        run1 = client.create_run(exp_id)
        client.log_param(run1.info.run_id, "model", "xgb")
        client.set_terminated(run1.info.run_id, "FINISHED")

        run2 = client.create_run(exp_id)
        client.log_param(run2.info.run_id, "model", "rf")
        client.set_terminated(run2.info.run_id, "FINISHED")

        runs = client.search_runs(
            experiment_ids=[exp_id],
            filter_string="params.model = 'xgb'",
        )
        assert len(runs) == 1
        assert runs[0].info.run_id == run1.info.run_id

    def test_search_runs_order_by_metric(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-search-order")
        for val in [0.1, 0.9, 0.5]:
            run = client.create_run(exp_id)
            client.log_metric(run.info.run_id, "score", val)
            client.set_terminated(run.info.run_id, "FINISHED")

        runs = client.search_runs(
            experiment_ids=[exp_id],
            order_by=["metrics.score DESC"],
        )
        scores = [r.data.metrics["score"] for r in runs]
        assert scores == sorted(scores, reverse=True)

    def test_search_runs_deleted_view(self, client: MlflowClient):
        exp_id = client.create_experiment("e2e-search-deleted")
        run = client.create_run(exp_id)
        client.set_terminated(run.info.run_id, "FINISHED")
        client.delete_run(run.info.run_id)

        # Active view should not include deleted
        active = client.search_runs(experiment_ids=[exp_id], run_view_type=ViewType.ACTIVE_ONLY)
        assert all(r.info.run_id != run.info.run_id for r in active)

        # Deleted view should include it
        deleted = client.search_runs(experiment_ids=[exp_id], run_view_type=ViewType.DELETED_ONLY)
        assert any(r.info.run_id == run.info.run_id for r in deleted)


class TestSearchExperimentsUI:
    """Test the exact search_experiments calls the MLflow UI makes."""

    def test_ui_homepage_call(self, client: MlflowClient):
        """The MLflow UI homepage calls this on load."""
        experiments = client.search_experiments(
            max_results=25,
            order_by=["last_update_time DESC"],
            filter_string="tags.`mlflow.experiment.isGateway` IS NULL",

## Implementation Notes

### Fixing Decimal serialization

When e2e tests fail with `TypeError: 'decimal.Decimal' object cannot be interpreted as an integer`, the fix is in the entity conversion functions. Every DynamoDB number comes back as `decimal.Decimal`. All timestamp fields must be cast to `int()`:

**`tracking_store.py`:**
```python
def _item_to_experiment(item, tags=None):
    return Experiment(
        ...
        creation_time=int(item["creation_time"]) if item.get("creation_time") is not None else None,
        last_update_time=int(item["last_update_time"]) if item.get("last_update_time") is not None else None,
    )

def _item_to_run_info(item):
    return RunInfo(
        ...
        start_time=int(item["start_time"]) if item.get("start_time") is not None else None,
        end_time=int(item["end_time"]) if item.get("end_time") is not None else None,
    )
```

**`registry_store.py`:**
```python
def _item_to_registered_model(item, tags=None):
    return RegisteredModel(
        ...
        creation_timestamp=int(item["creation_timestamp"]) if item.get("creation_timestamp") is not None else None,
        last_updated_timestamp=int(item["last_updated_timestamp"]) if item.get("last_updated_timestamp") is not None else None,
    )

def _item_to_model_version(item, tags=None):
    return ModelVersion(
        ...
        creation_timestamp=int(item["creation_timestamp"]) if item.get("creation_timestamp") is not None else 0,
        last_updated_timestamp=int(item["last_updated_timestamp"]) if item.get("last_updated_timestamp") is not None else None,
    )
```

A helper function may be cleaner:
```python
def _int_or_none(value) -> int | None:
    return int(value) if value is not None else None
```

### Server startup

The mlflow server fixture needs to handle:
- Port allocation (use `--port 0` or find a free port)
- Waiting for the server to be ready (parse stdout or poll health endpoint)
- Graceful shutdown on test completion
- Single worker (`--workers 1`) to avoid race conditions in tests

### Test isolation

The server is session-scoped (started once). Tests should use unique experiment/model names to avoid collisions. If stronger isolation is needed, consider function-scoped fixtures that create unique experiments.
