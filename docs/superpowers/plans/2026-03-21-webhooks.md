> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement webhook CRUD operations in `DynamoDBRegistryStore` to pass all 21 compatibility tests (20 registry + 1 workspace) plus a new unit test for `list_webhooks_by_event`.

**Architecture:** Each webhook is stored as a partition with a META item and N EVENT items. GSI2 indexes webhooks by workspace for listing; GSI3 indexes events for `list_webhooks_by_event`. Secrets are Fernet-encrypted using `MLFLOW_WEBHOOK_SECRET_ENCRYPTION_KEY`. Soft delete removes GSI keys and deletes EVENT items.

**Tech Stack:** Python, DynamoDB (moto for tests), cryptography (Fernet), MLflow entities/validation

**Spec:** `docs/superpowers/specs/2026-03-20-webhooks-design.md`

```python
def _get_fernet() -> Fernet:
    """Get a Fernet cipher for webhook secret encryption/decryption."""
    key = MLFLOW_WEBHOOK_SECRET_ENCRYPTION_KEY.get()
    if key is None:
        key = Fernet.generate_key()
    if isinstance(key, str):
        key = key.encode()
    return Fernet(key)


def _encrypt_webhook_secret(secret: str) -> str:
    """Encrypt a webhook secret using Fernet."""
    return _get_fernet().encrypt(secret.encode()).decode()


def _decrypt_webhook_secret(encrypted: str) -> str:
    """Decrypt a webhook secret using Fernet."""
    return _get_fernet().decrypt(encrypted.encode()).decode()
```

```python
def _get_webhook_items(self, webhook_ulid: str) -> list[dict]:
    """Query all items in a webhook partition (META + EVENT items)."""
    return self._table.query(
        pk=f"{PK_WEBHOOK_PREFIX}{webhook_ulid}",
    )

def _webhook_items_to_entity(self, webhook_ulid: str, items: list[dict]) -> Webhook:
    """Convert raw DynamoDB items from a webhook partition to a Webhook entity.

    Classifies items by SK prefix (WH#META vs WH#EVT#), decrypts secret,
    and assembles the Webhook entity.
    """
    meta: dict | None = None
    events: list[WebhookEvent] = []

    for item in items:
        sk = item["SK"]
        if sk == SK_WEBHOOK_META:
            meta = item
        elif sk.startswith(SK_WEBHOOK_EVT_PREFIX):
            events.append(
                WebhookEvent(
                    entity=item["entity"],
                    action=item["action"],
                )
            )

    if meta is None:
        raise MlflowException(
            f"Webhook with ID {webhook_ulid} not found",
            error_code=RESOURCE_DOES_NOT_EXIST,
        )

    # Decrypt secret if present
    secret = None
    if meta.get("encrypted_secret"):
        secret = _decrypt_webhook_secret(meta["encrypted_secret"])

    return Webhook(
        webhook_id=webhook_ulid,
        name=meta["name"],
        url=meta["url"],
        events=events,
        description=meta.get("description"),
        status=WebhookStatus(meta["status"]),
        secret=secret,
        creation_timestamp=int(meta["creation_timestamp"]),
        last_updated_timestamp=int(meta["last_updated_timestamp"]),
        workspace=meta.get("workspace"),
    )

def _get_webhook_by_id(self, webhook_id: str) -> Webhook:
    """Get a webhook by ID. Raises RESOURCE_DOES_NOT_EXIST if not found or soft-deleted."""
    items = self._get_webhook_items(webhook_id)
    if not items:
        raise MlflowException(
            f"Webhook with ID {webhook_id} not found",
            error_code=RESOURCE_DOES_NOT_EXIST,
        )

    # Find META item and check for soft delete
    meta = None
    for item in items:
        if item["SK"] == SK_WEBHOOK_META:
            meta = item
            break

    if meta is None or meta.get("deleted_timestamp") is not None:
        raise MlflowException(
            f"Webhook with ID {webhook_id} not found",
            error_code=RESOURCE_DOES_NOT_EXIST,
        )

    # Filter to only workspace-matching items
    if meta.get("workspace") != self._workspace:
        raise MlflowException(
            f"Webhook with ID {webhook_id} not found",
            error_code=RESOURCE_DOES_NOT_EXIST,
        )

    return self._webhook_items_to_entity(webhook_id, items)
```

```python
def create_webhook(
    self,
    name: str,
    url: str,
    events: list[WebhookEvent],
    description: str | None = None,
    secret: str | None = None,
    status: WebhookStatus | None = None,
) -> Webhook:
    _validate_webhook_name(name)
    _validate_webhook_url(url)
    _validate_webhook_events(events)

    webhook_ulid = generate_ulid()
    now = get_current_time_millis()
    ws = self._workspace
    effective_status = status or WebhookStatus.ACTIVE

    # Build META item
    meta_item: dict = {
        "PK": f"{PK_WEBHOOK_PREFIX}{webhook_ulid}",
        "SK": SK_WEBHOOK_META,
        "name": name,
        "url": url,
        "status": effective_status.value,
        "creation_timestamp": now,
        "last_updated_timestamp": now,
        "workspace": ws,
        # GSI2 for list_webhooks
        GSI2_PK: f"{GSI2_WEBHOOKS_PREFIX}{ws}",
        GSI2_SK: webhook_ulid,
    }
    if description is not None:
        meta_item["description"] = description
    if secret is not None:
        meta_item["encrypted_secret"] = _encrypt_webhook_secret(secret)

    # Build EVENT items
    event_items = []
    for event in events:
        event_items.append({
            "PK": f"{PK_WEBHOOK_PREFIX}{webhook_ulid}",
            "SK": f"{SK_WEBHOOK_EVT_PREFIX}{event.entity.value}#{event.action.value}",
            "entity": event.entity.value,
            "action": event.action.value,
            "workspace": ws,
            # GSI3 for list_webhooks_by_event
            GSI3_PK: f"{GSI3_WH_EVT_PREFIX}{ws}#{event.entity.value}#{event.action.value}",
            GSI3_SK: webhook_ulid,
        })

    # Write all items
    self._table.batch_write([meta_item] + event_items)

    return Webhook(
        webhook_id=webhook_ulid,
        name=name,
        url=url,
        events=events,
        description=description,
        status=effective_status,
        secret=secret,
        creation_timestamp=now,
        last_updated_timestamp=now,
        workspace=ws,
    )
```

```python
def list_webhooks(
    self,
    max_results: int | None = None,
    page_token: str | None = None,
) -> PagedList[Webhook]:
    max_results = max_results or 100
    if max_results < 1 or max_results > 1000:
        raise MlflowException(
            "max_results must be between 1 and 1000.",
            INVALID_PARAMETER_VALUE,
        )

    exclusive_start_key = self._decode_page_token(page_token)

    # Query GSI2 for all webhooks in this workspace (reverse order for DESC)
    # Use query_page (not query) for cursor-based pagination
    gsi2_items, lek = self._table.query_page(
        pk=f"{GSI2_WEBHOOKS_PREFIX}{self._workspace}",
        index_name=GSI2_NAME,
        limit=max_results,
        scan_forward=False,
        exclusive_start_key=exclusive_start_key,
    )

    # Build next page token from DynamoDB's LastEvaluatedKey
    next_page_token = None
    if lek is not None:
        next_page_token = self._encode_page_token(lek)

    # Fetch full partition for each webhook
    webhooks = []
    for gsi2_item in gsi2_items:
        webhook_ulid = gsi2_item[GSI2_SK]
        items = self._get_webhook_items(webhook_ulid)
        if items:
            webhooks.append(self._webhook_items_to_entity(webhook_ulid, items))

    return PagedList(webhooks, next_page_token)
```

```python
def update_webhook(
    self,
    webhook_id: str,
    name: str | None = None,
    description: str | None = None,
    url: str | None = None,
    events: list[WebhookEvent] | None = None,
    secret: str | None = None,
    status: WebhookStatus | None = None,
) -> Webhook:
    # Validate provided fields
    if name is not None:
        _validate_webhook_name(name)
    if url is not None:
        _validate_webhook_url(url)
    if events is not None:
        _validate_webhook_events(events)

    # Verify webhook exists (also checks workspace + not deleted)
    existing = self._get_webhook_by_id(webhook_id)

    now = get_current_time_millis()
    updates: dict = {"last_updated_timestamp": now}

    if name is not None:
        updates["name"] = name
    if description is not None:
        updates["description"] = description
    if url is not None:
        updates["url"] = url
    if secret is not None:
        updates["encrypted_secret"] = _encrypt_webhook_secret(secret)
    if status is not None:
        updates["status"] = status.value

    # Update META item
    self._table.update_item(
        pk=f"{PK_WEBHOOK_PREFIX}{webhook_id}",
        sk=SK_WEBHOOK_META,
        updates=updates,
    )

    # If events changed, replace all EVENT items
    if events is not None:
        ws = self._workspace

        # Delete old EVENT items
        old_items = self._table.query(
            pk=f"{PK_WEBHOOK_PREFIX}{webhook_id}",
            sk_prefix=SK_WEBHOOK_EVT_PREFIX,
        )
        if old_items:
            self._table.batch_delete(
                [{"PK": item["PK"], "SK": item["SK"]} for item in old_items]
            )

        # Write new EVENT items
        new_event_items = []
        for event in events:
            new_event_items.append({
                "PK": f"{PK_WEBHOOK_PREFIX}{webhook_id}",
                "SK": f"{SK_WEBHOOK_EVT_PREFIX}{event.entity.value}#{event.action.value}",
                "entity": event.entity.value,
                "action": event.action.value,
                "workspace": ws,
                GSI3_PK: f"{GSI3_WH_EVT_PREFIX}{ws}#{event.entity.value}#{event.action.value}",
                GSI3_SK: webhook_id,
            })
        if new_event_items:
            self._table.batch_write(new_event_items)

    # Re-fetch and return updated webhook
    return self._get_webhook_by_id(webhook_id)
```

```python
def delete_webhook(self, webhook_id: str) -> None:
    # Verify webhook exists (also checks workspace + not deleted)
    self._get_webhook_by_id(webhook_id)

    now = get_current_time_millis()
    updates: dict = {
        "deleted_timestamp": now,
        "last_updated_timestamp": now,
    }

    # Optional TTL for eventual hard-delete
    ttl_seconds = self._config.get_soft_deleted_ttl_seconds()
    if ttl_seconds is not None:
        import time
        updates["ttl"] = int(time.time()) + ttl_seconds

    # Remove GSI2 keys so webhook disappears from list_webhooks
    removes = [GSI2_PK, GSI2_SK]

    self._table.update_item(
        pk=f"{PK_WEBHOOK_PREFIX}{webhook_id}",
        sk=SK_WEBHOOK_META,
        updates=updates,
        removes=removes,
    )

    # Hard-delete all EVENT items (disappears from list_webhooks_by_event)
    event_items = self._table.query(
        pk=f"{PK_WEBHOOK_PREFIX}{webhook_id}",
        sk_prefix=SK_WEBHOOK_EVT_PREFIX,
    )
    if event_items:
        self._table.batch_delete(
            [{"PK": item["PK"], "SK": item["SK"]} for item in event_items]
        )
```

```python
def list_webhooks_by_event(
    self,
    event: WebhookEvent,
    max_results: int | None = None,
    page_token: str | None = None,
) -> PagedList[Webhook]:
    max_results = max_results or 100
    if max_results < 1 or max_results > 1000:
        raise MlflowException(
            "max_results must be between 1 and 1000.",
            INVALID_PARAMETER_VALUE,
        )

    exclusive_start_key = self._decode_page_token(page_token)
    ws = self._workspace

    # Query GSI3 for webhooks subscribed to this specific event
    # Use query_page (not query) for cursor-based pagination
    gsi3_items, lek = self._table.query_page(
        pk=f"{GSI3_WH_EVT_PREFIX}{ws}#{event.entity.value}#{event.action.value}",
        index_name=GSI3_NAME,
        limit=max_results,
        scan_forward=False,
        exclusive_start_key=exclusive_start_key,
    )

    # Build next page token from DynamoDB's LastEvaluatedKey
    next_page_token = None
    if lek is not None:
        next_page_token = self._encode_page_token(lek)

    # Fetch full partition for each webhook to build complete entity
    webhooks = []
    for gsi3_item in gsi3_items:
        webhook_ulid = gsi3_item[GSI3_SK]
        items = self._get_webhook_items(webhook_ulid)
        if items:
            webhooks.append(self._webhook_items_to_entity(webhook_ulid, items))

    return PagedList(webhooks, next_page_token)
```
